```ts
// qa.prompt.ts
import type { RetrievedSource } from '../gateway/provider.types.js';
interface RetrievedChunk { objectType:string; objectId:string; scopeType:string; scopeId:string; content:string; score:number; }

export function buildAskPrompt(question: string, chunks: RetrievedChunk[]):
    { prompt: string; sources: RetrievedSource[] } {
  const sources: RetrievedSource[] = chunks.map((c, i) =>
    ({ id: String(i + 1), objectType: c.objectType, objectId: c.objectId, content: c.content }));
  const ctx = sources.map(s => `[${s.id}] (${s.objectType}) ${s.content}`).join('\n');
  const prompt =
    `Answer the question using ONLY the numbered sources. Cite sources inline as [n]. ` +
    `If the sources don't contain the answer, say so.\n\nSources:\n${ctx}\n\nQuestion: ${question}`;
  return { prompt, sources };
}

export function parseCitations(answer: string, sources: RetrievedSource[]):
    { objectType: string; objectId: string }[] {
  const cited = new Set([...answer.matchAll(/\[(\d+)\]/g)].map(m => m[1]));
  return sources.filter(s => cited.has(s.id)).map(s => ({ objectType: s.objectType, objectId: s.objectId }));
}
```

- [ ] **Step 4: Run — PASS. Step 5: Commit.** `feat(11b): pure ask-prompt builder + citation parser`

### Task 2: `qa.service.ts` — orchestration

**Files:** Create `apps/api/src/modules/ai/qa/qa.service.ts`

- [ ] **Step 1: Implement** (citations derived ONLY from retrieved chunks — spec §4.3):

```ts
export class QaService {
  constructor(private retrieval = retrievalService, private gateway = aiGatewayService) {}
  async ask(userId: string, workspaceId: string, question: string, scope?: {type:string;id:string}) {
    const chunks = await this.retrieval.retrieve(userId, workspaceId, question, { scope, k: 8 });
    const { prompt, sources } = buildAskPrompt(question, chunks);
    const { text } = await this.gateway.complete({ workspaceId, userId, feature: 'qa' }, { prompt, sources });
    const citations = parseCitations(text, sources); // ⊆ already-permission-filtered chunks
    return { answer: text, citations };
  }
}
```
